202    /**
203     * Change de salle
204     * 
205     * @param pCommand Commande de changement de salle
206     */
207    private void goRoom(final Command pCommand) {
208        if (!pCommand.hasSecondWord()) {
209            this.aGui.println("Go Where ?");
210            return;
211        }
212
213        if (!CommandWords.isDirection(pCommand.getSecondWord())) {
214            this.aGui.println("Unknown direction !");
215            return;
216        }
217
218        String vDirection = pCommand.getSecondWord();
219
220        if (this.aPlayer.getCurrentRoom().getExit(vDirection) == null) {
221            this.aGui.println("There is no door !");
222            return;
223        }
224
225        this.aPlayer.goRoom(vDirection);
226        this.printLocationInfo();
227    }

257    /**
258     * Given a command, process (that is: execute) the command.
259     * 
260     * @param pCommandLine The command to be processed.
261     */
262    public void interpretCommand(final String pCommandLine) {
263        this.aGui.println("> " + pCommandLine);
264        Command vCommand = this.aParser.getCommand(pCommandLine);
265
266        if (vCommand.isUnknown()) {
267            this.aGui.println("I don't know what you mean...");
268            return;
269        }
270
271        String vCommandWord = vCommand.getCommandWord();
272        vCommandWord = vCommandWord.toLowerCase();
273        switch (vCommandWord) {
274            case "help":
275                this.printHelp();
276                break;
277            case "look":
278                this.look();
279                break;
280            case "pray":
281                this.pray();
282                break;
283            case "go":
284                this.goRoom(vCommand);
285                break;
286            case "back":
287                this.back();
288                break;
289            case "take":
290                this.take(vCommand);
291                break;
292            case "drop":
293                this.drop(vCommand);
294                break;
295            case "quit":
296                if (vCommand.hasSecondWord())
297                    this.aGui.println("Quit what?");
298                else
299                    this.endGame();
300                break;
301            case "items":
302                this.printItems();
303                break;
304            case "eat":
305                this.eat(vCommand);
306                break;
307            case "test":
308                this.test(vCommand);
309                break;
310            case "respawn":
311                this.respawn();
312                break;
313            default:
314                this.aGui.println("Command not implemented yet");
315                break;
316        }
317
318        this.runMechanics();
319    }

408

413
