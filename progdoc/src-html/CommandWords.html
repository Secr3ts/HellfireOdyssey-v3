001 /**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * 
005 * This class holds an enumeration table of all command words known to the game.
006 * It is used to recognise commands as they are typed in.
007 *
008 * @author  Michael Kolling and David J. Barnes + D.Bureau + A.Fournier
009 * @version 2008.03.30 + 2019.09.25 + 2023.04.18
010 */
011public class CommandWords
012{
013     // a constant array that will hold all valid command words
014    private final String[] aValidCommands;
015    private static final String[] aValidDirection = {"north", "south", "east", "west", "up", "down"};
016    /**
017     * Constructor - initialise the command words.
018     */
019    public CommandWords()
020    {
021        this.aValidCommands = new String[]{
022            "go",
023            "help",
024            "quit",
025            "look",
026            "pray",
027            "back",
028            "take",
029            "drop",
030            "test",
031            "respawn"
032        };
033    } // CommandWords()
034    /**
035     * Check whether a given String is a valid direction word.
036     * @param pString
037     * @return true if a given string is a valid direction, false otherwise.
038     */
039    public static boolean isDirection(final String pString) {
040        for (int vI = 0; vI < aValidDirection.length; vI++) {
041            if (aValidDirection[vI].equals(pString)) {
042                return true;
043            }
044        }
045        return false;
046    }
047    /**
048     * Check whether a given String is a valid command word. 
049     * @return true if a given string is a valid command,
050     * false if it isn't.
051     */
052    public boolean isCommand( final String pString )
053    {
054        for ( int vI=0; vI<this.aValidCommands.length; vI++ ) {
055            if ( this.aValidCommands[vI].equals( pString ) )
056                return true;
057        } // for
058        // if we get here, the string was not found in the commands :
059        return false;
060    } // isCommand()
061
062    /**
063     * @return a string of all valid commands
064     */
065    public String[] getCommands() {
066        return this.aValidCommands;
067    }
068    /**
069     * @return a string of all valid commands
070     */
071    public String getCommandList() {
072        String vReturn = "";
073        for (int vI = 0; vI < this.aValidCommands.length; vI++) {
074            vReturn += this.aValidCommands[vI] + " ";
075        }
076        return vReturn;
077    }
078} // CommandWords
