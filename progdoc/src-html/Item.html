001/**
002 * This class represents an item in the game.
003 */
004public class Item {
005    private String aName;
006    private String aDescription;
007    private int aWorth;
008    private int aWeight;
009
010    /**
011     * Constructor for objects of class Item
012     * @param pName the name of the item
013     * @param pDescription the description of the item
014     * @param pWorth the worth of the item
015     * @param pWeight the weight of the item
016     */
017    public Item(final String pName, final String pDescription, final int pWorth, final int pWeight) {
018        this.aName = pName;
019        this.aDescription = pDescription;
020        this.aWorth = pWorth;
021        this.aWeight = pWeight;
022    }
023    
024    /**
025     * @return the description of the item
026     */
027    public String getDescription() {
028        return this.aDescription;
029    }
030
031    /**
032     * @return the name of the item
033     */
034    public String getName() {
035        return this.aName;
036    }
037    /**
038     * @return the worth of the item
039     */
040    public int getWorth() {
041        return this.aWorth;
042    }
043
044    /**
045     * @return the weight of the item
046     */
047    public int getWeight() {
048        return this.aWeight;
049    }
050}
